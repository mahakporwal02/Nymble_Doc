603 void Mat::forEach_impl(const Functor& operation) {
604  if (false) {
605  operation(*reinterpret_cast<_Tp*>(0), reinterpret_cast<int*>(0));
606  // If your compiler fails in this line.
607  // Please check that your functor signature is
608  // (_Tp&, const int*) <- multi-dimensional
609  // or (_Tp&, void*) <- in case you don't need current idx.
610  }

615  class PixelOperationWrapper :public ParallelLoopBody
616  {
617  public:
618  PixelOperationWrapper(Mat_<_Tp>* const frame, const Functor& _operation)
619  : mat(frame), op(_operation) {}
620  virtual ~PixelOperationWrapper(){}
621  // ! Overloaded virtual operator
622  // convert range call to row call.
623  virtual void operator()(const Range &range) const CV_OVERRIDE
624  {
625  const int DIMS = mat->dims;
626  const int COLS = mat->size[DIMS - 1];
627  if (DIMS <= 2) {
628  for (int row = range.start; row < range.end; ++row) {
629  this->rowCall2(row, COLS);
630  }
631  } else {
632  std::vector<int> idx(DIMS);
633  idx[DIMS - 2] = range.start - 1;
634 
635  for (int line_num = range.start; line_num < range.end; ++line_num) {
636  idx[DIMS - 2]++;
637  for (int i = DIMS - 2; i >= 0; --i) {
638  if (idx[i] >= mat->size[i]) {
639  idx[i - 1] += idx[i] / mat->size[i];
640  idx[i] %= mat->size[i];
641  continue; // carry-over;
642  }
643  else {
644  break;
645  }
646  }
647  this->rowCall(&idx[0], COLS, DIMS);
648  }
649  }
650  }
651  private:
652  Mat_<_Tp>* const mat;
653  const Functor op;
654  // ! Call operator for each elements in this row.
655  inline void rowCall(int* const idx, const int COLS, const int DIMS) const {
656  int &col = idx[DIMS - 1];
657  col = 0;
658  _Tp* pixel = &(mat->template at<_Tp>(idx));
659 
660  while (col < COLS) {
661  op(*pixel, const_cast<const int*>(idx));
662  pixel++; col++;
663  }
664  col = 0;
665  }
666  // ! Call operator for each elements in this row. 2d mat special version.
667  inline void rowCall2(const int row, const int COLS) const {
668  union Index{
669  int body[2];
670  operator const int*() const {
671  return reinterpret_cast<const int*>(this);
672  }
673  int& operator[](const int i) {
674  return body[i];
675  }
676  } idx = {{row, 0}};
677  // Special union is needed to avoid
678  // "error: array subscript is above array bounds [-Werror=array-bounds]"
679  // when call the functor `op` such that access idx[3].
680 
681  _Tp* pixel = &(mat->template at<_Tp>(idx));
682  const _Tp* const pixel_end = pixel + COLS;
683  while(pixel < pixel_end) {
684  op(*pixel++, static_cast<const int*>(idx));
685  idx[1]++;
686  }
687  }
688  PixelOperationWrapper& operator=(const PixelOperationWrapper &) {
689  CV_Assert(false);
690  // We can not remove this implementation because Visual Studio warning C4822.
691  return *this;
692  }
693  };
694 
695  parallel_for_(cv::Range(0, LINES), PixelOperationWrapper(reinterpret_cast<Mat_<_Tp>*>(this), operation));
696 }

1109  {

1111  {
1112  if(this->m_childs[i]->m_payload == payload)
1113  return this->m_childs[i];

1115  return NULL;
1116  }

1118  int findChild(Node<OBJECT> *pNode) const
1119  {
1120  for (size_t i = 0; i < this->m_childs.size(); i++)
1121  {
1122  if(this->m_childs[i] == pNode)
1123  return (int)i;
1124  }
1125  return -1;
1126  }

1129  {
1130  if(!pNode)
1131  return;

1133  CV_Assert(pNode->m_pParent == 0);
1134  pNode->m_pParent = this;
1135  this->m_childs.push_back(pNode);

1141  {

1143  delete m_childs[i];
1144  }
1145  m_childs.clear();
1146  }
