117 float AdcRequestProcessor::ProcessAdcRequest(AdcRequest adc_request) {
118  Component_e hardware_component = adc_request.GetHardwareComponent();
119  Adc *adc = AdcRequestProcessor::component_to_device_map_.at(hardware_component);
120  float result;
121  if (adc->getDeviceId() == DEFAULT_ADC_ENUM) {
122  return adc->ReadChannel(0);
123  } else {
124  switch (adc->getDeviceId()) {
125  case ADC_INA219B:
126  case ADC_ADS122C04: {
127  if (adc_request.GetAdcRequestType() == REFERENCE_REQUEST) {
128  result = adc->GetAdcFactor();
129  } else {
130  result = adc->ReadChannel(0);
131  }
132  }
133  break;
134  case ADC_ADS1115: {
135  Ads1115Channels_e channel = AdcRequestProcessor::component_to_ads1115Channel_map_.at(
136  hardware_component);
137  result = ads1115_->ReadChannel(channel);
138  }
139  break;
140  case ADC_MCP3426: {
141  MCP3462Channels_e channel = AdcRequestProcessor::component_to_mcp3426_map_.at(
142  hardware_component);
143  result = mcp3426_->ReadChannel(channel);
144  }
145  break;
146  case ADC_MAX11607: {
147  Max11607Channels_e channel = AdcRequestProcessor::component_to_max11607_channel_map_.at(
148  hardware_component);
149  result = max11607_->ReadChannel(channel);
150  }
151  break;
152  default: {
153  __android_log_print(ANDROID_LOG_ERROR, "Chef : ",
154  "Adc error %d",
155  adc->getDeviceId());
156  throw new Error;
157  }
158  }
159  }
160  return result;
161 }
