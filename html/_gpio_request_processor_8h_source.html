186 int GpioRequestProcessor::ProcessGpioRequest(GpioRequest gpio_request) {
187  Component_e hardware_component = gpio_request.GetHardwareComponent();
188  Gpio *gpio = GpioRequestProcessor::component_to_device_map_.at(hardware_component);
189  int result = -1;
190  switch (gpio->getDeviceId()) {
191  case GPIO_PCA9535: {
192  Pca9535Channels_e pca_9535_pin = GpioRequestProcessor::component_to_pca_channel_map_.at(
193  hardware_component);
194  switch (gpio_request.GetAction()) {
195  case ACTION_READ_PIN: {
196  result = GpioRequestProcessor::pca9535_->ReadPinState(pca_9535_pin);
197  }
198  break;
199  case ACTION_WRITE_PIN: {
200  result = GpioRequestProcessor::pca9535_->SetPinState(pca_9535_pin,
201  gpio_request.GetState());
202  }
203  break;
204  case ACTION_SET_DIRECTION: {
205  result = GpioRequestProcessor::pca9535_->SetPinDirection(pca_9535_pin,
206  gpio_request.GetDirection());
207  }
208  default: {
209  throw new Error;
210  }
211  }
212  }
213  break;
214  case DEFAULT_GPIO_DEVICE: {
215  SomGpioPins_e som_pin = GpioRequestProcessor::component_to_som_channel_map_.at(
216  hardware_component);
217  switch (gpio_request.GetAction()) {
218  case ACTION_READ_PIN: {
219  result = GpioRequestProcessor::som_gpio_->ReadPinState(som_pin);
220  }
221  break;
222  case ACTION_WRITE_PIN: {
223  if (gpio_request.GetState() == STATE_DEFAULT) {
224  throw new Error;
225  }
226  result = GpioRequestProcessor::som_gpio_->SetPinState(som_pin,
227  gpio_request.GetState());
228  }
229  break;
230  case ACTION_SET_DIRECTION: {
231  if (gpio_request.GetDirection() == GPIO_DIRECTION_DEFAULT) {
232  throw new Error;
233  }
234  result = GpioRequestProcessor::som_gpio_->SetPinDirection(som_pin,
235  gpio_request.GetDirection());
236  }
237  break;
238  default: {
239  throw new Error;
240  }
241  }
242  }
243  break;
244  default: {
245  throw new Error;
246  }
247  }
248  return result;
249 }

251 void GpioRequestProcessor::InitializeGpioPins() {
252  for (auto itr = GpioRequestProcessor::component_to_default_direction_map_.begin();
253  itr != GpioRequestProcessor::component_to_default_direction_map_.end(); itr++) {
254  Component_e hardware_component = itr->first;
255  GpioDirection_e direction = itr->second;
256  Gpio *gpio = GpioRequestProcessor::component_to_device_map_.at(hardware_component);
257 
258  if (GpioRequestProcessor::component_to_device_map_.at(itr->first)->getDeviceId() ==
259  GPIO_PCA9535) {
260  gpio->SetPinDirection(
261  GpioRequestProcessor::component_to_pca_channel_map_.at(hardware_component),
262  direction);
263  if (GpioRequestProcessor::component_to_default_direction_map_.at(hardware_component) ==
264  DIRECTION_OUTPUT) {
265  gpio->SetPinState(
266  GpioRequestProcessor::component_to_pca_channel_map_.at(hardware_component),
267  GpioRequestProcessor::component_to_default_state_map_.at(
268  hardware_component));
269  }
270  } else {
271  gpio->SetPinDirection(
272  GpioRequestProcessor::component_to_som_channel_map_.at(hardware_component),
273  direction);
274  if (GpioRequestProcessor::component_to_default_direction_map_.at(hardware_component) ==
275  DIRECTION_OUTPUT) {
276  gpio->SetPinState(
277  GpioRequestProcessor::component_to_som_channel_map_.at(hardware_component),
278  GpioRequestProcessor::component_to_default_state_map_.at(
279  hardware_component));
280  }
281  }
282 
283  }
