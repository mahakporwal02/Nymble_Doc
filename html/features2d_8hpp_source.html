790 struct CV_EXPORTS Accumulator
791 {
792  typedef T Type;

800 /*
801  * Squared Euclidean distance functor
802  */
803 template<class T>
804 struct CV_EXPORTS SL2
805 {
806  static const NormTypes normType = NORM_L2SQR;
807  typedef T ValueType;
808  typedef typename Accumulator<T>::Type ResultType;
809 
810  ResultType operator()( const T* a, const T* b, int size ) const
811  {
812  return normL2Sqr<ValueType, ResultType>(a, b, size);
813  }
814 };

816 /*
817  * Euclidean distance functor
818  */
819 template<class T>
820 struct L2
821 {
822  static const NormTypes normType = NORM_L2;
823  typedef T ValueType;
824  typedef typename Accumulator<T>::Type ResultType;
825 
826  ResultType operator()( const T* a, const T* b, int size ) const
827  {
828  return (ResultType)std::sqrt((double)normL2Sqr<ValueType, ResultType>(a, b, size));
829  }
830 };
