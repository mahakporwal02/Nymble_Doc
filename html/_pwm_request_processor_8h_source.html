135 int PwmRequestProcessor::ProcessPwmRequest(PwmRequest pwmRequest) {
136  I2cDevices_e device_id = PwmRequestProcessor::component_to_device_map_.at(
137  pwmRequest.GetHardwareComponent())->GetDeviceId();
138  int result = -1;
139  switch (device_id) {
140  case PWM_PCA9685: {
141  Pca9685Channels_e channel = PwmRequestProcessor::component_to_pca_channel_map_.at(
142  pwmRequest.GetHardwareComponent());
143  if (pwmRequest.GetDutyCycle() != -1) {
144  result = PwmRequestProcessor::pca9685_->SetDutyCycle(channel,
145  pwmRequest.GetDutyCycle());
146  } else if (pwmRequest.GetPulseDurationUs() != -1) {
147  float duty_cycle = 0.0;
148  duty_cycle = (pwmRequest.GetPulseDurationUs() / 200.0);
149  __android_log_print(ANDROID_LOG_ERROR, "Chef : ",
150  "pca dutycycle %.2f", duty_cycle);
151  result = PwmRequestProcessor::pca9685_->SetDutyCycle(channel, duty_cycle);
152  } else {
153  __android_log_print(ANDROID_LOG_ERROR, "Chef : ",
154  "pwm processor no duty cycle and no pulse duration ");
155  throw (new Error);
156  }
157  break;
158  }
159  case PWM_DS1050Z_INDUCTION:
160  case PWM_DS1050Z_LED_ILLUMINATION:
161  case PWM_DS1050Z_STIRRER:
162  case PWM_MAX6650:
163  case PWM_DS1050Z_EXHAUST: {
164  PwmGenerator *pwm_generator = PwmRequestProcessor::component_to_device_map_.at(
165  pwmRequest.GetHardwareComponent());
166  result = pwm_generator->SetDutyCycle(0, pwmRequest.GetDutyCycle());
167  break;
168  }
169  default:
170  break;
171  }
172 
173  return result;
174 }
